按月查看总体销售额
select months, sum(pro_sum) 
from work.sales 
group by months;

按月查看新增注册人数
select months,users_new_register
from work.users_allinfo 
order by months;

按月查看城市销售额（收货城市）
select months,city,pro_sum
from work.sale_city;
order by months;

按月查看城市、商品类别的销售额（两个维度：城市，商品类别）
select work.sale_city.months,city,category, work.sale_city.pro_sum
from work.sale_city,work.sale_cate 
where work.sale_city.months = work.sale_cate.months;

按月查看性别、商品类别的销售额（）
select work.sale_sex.months,sex,category,work.sale_sex.pro_sum
from work.sale_sex,work.sale_cate 
where work.sale_sex.months = work.sale_cate.months;
